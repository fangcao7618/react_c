```
  UEdtiorConfigsMyPlugin 配置项
  UEdtiorConfigsMyPlugin = {
    image: 'image'|'embed'|'video',
    embed: 'image'|'embed'|'video',
    video: 'image'|'embed'|'video'
  };

  // 这个代表原编辑器中 image 展示的用 embed 来展示
  window.UEdtiorConfigsMyPlugin.image = 'image';

  // 这个代表原本 video 的展示用 embed 来展示
  window.UEdtiorConfigsMyPlugin.video = 'embed';

```
